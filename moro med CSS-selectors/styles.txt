/*tagger h1 blir arial i gul farge*/
h| {
	 font-family:Arial;
	 color:yellow;
}

/*med klassen "important" settes fontfargen til rød*/
.important {
					 color:red;
					 
}

/*med id "somesuch" settes fontfargen til blå*/
#somesuch {
					color:blue;
}

/*Alle span som heter "important" får mørkerød fontfarge*/
span.important {
							 color:darkred;
}

/*overskrift 1, 2 og 3, og paragraf klasser merket "important" flyttes fra venstre 50 pixler*/
h1, h2, h3, p.important {
		margin-left:50px;
}

/*lenker blir grønne, og ikke uthevet på noen måte*/
a:link {
			 text-decoration:none;
			 color:green;
}

/*når pila ligger over lenken, får lenken underlinje*/
a:hover {
					 text-decoration:underline;
}

/*de lenkene som er besøkt, får fontfarge mørkegrå*/
a:visisted {
					 color:darkgrey;
}

/*første bokstav i paragrafen blir forstørret med 3%*/
p:first-letter {
							 font-size:3em;
}

/*før en section som er en article settes disse tegnene inn*/
article:before {
							 content: '************************'
}

/*første paragraf i en type (section osv) får 600 pixler margin fra høyre*/
p:first-of-type {
								margin-right:600px;
}

/*første list-item blir kursivert*/
li:first-child {
							 font-style:italic;
}

/*siste list-item blir uthevet*/
li:last-child {
							 font-weight:bold;
}

/*det 3. list-item blir understreket*/
li:nth-of-type(3) {
							 text-decoration:underline;
}

/*der taggen "title" forekommer i et span, blir fontfargen svart*/
span[title] {
						color:black;
}

/*der verdien i taggen "title" er "first idea", blir teksten streket ut*/
span[title="first idea"] {
								 text-decoration:line-through;
}
/*epostlenker blir forstørret med 2%*/
a[href^="email"] {
								 font-size:2em;
}
/*span som har tittel som ender på "idea" får strek i overkant*/
span[title$="idea"] {
										text-decoration:overline;
}
